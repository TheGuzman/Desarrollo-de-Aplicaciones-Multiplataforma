    # Descripción:      Este programa determina, de tres números
                        introducidos por el usuario, el más grande y lo imrprime por
                        pantalla

    # Parte declarativa:
        numero1: decimal
        numero2: decimal
        numero3: decimal

    # Cuerpo del algoritmo
        mostrarPorPantalla("Entre el valor d numero #1: ")
        leer numero1
        mostrarPorPantalla("Entre el valor d numero #2: ")
        leer numero2
        mostrarPorPantalla("Entre el valor d numero #3: ")
        leer numero3

        SI numero1 > numero2
            SI numero1 > numero3
                mostrarPorPantalla("El más grande es: ", numero1)
            SINO
                mostrarPorPantalla("El más grande es: ", numero3)
            FIN_SI
        SINO
            SI numero2 > numero3
                mostrarPorPantalla("El más grande es: ", numero2)
            SINO
                mostrarPorPantalla("El más grande es: ", numero3)
            FIN_SI
        FIN_SI
FIN
